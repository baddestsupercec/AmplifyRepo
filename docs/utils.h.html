              // Initialize WiFiClientSecure and PubSubClient objects
              WiFiClientSecure net = WiFiClientSecure();
              PubSubClient client(net);
              
              // Message handler function for incoming MQTT messages
              void messageHandler(char* topic, byte* payload, unsigned int length) {
                // Print incoming topic
                Serial.print("incoming: ");
                Serial.println(topic);
               
                // Deserialize JSON payload
                StaticJsonDocument<200> doc;
                deserializeJson(doc, payload);
                const char* message = doc["message"];
                Serial.println(message);
              }
              
              // Function to connect to AWS IoT Core
              void connectAWS() {
                // Print connection status message
                Serial.println("Connecting to Wi-Fi");
               
                // Wait for WiFi connection
                while (WiFi.status() != WL_CONNECTED){
                  delay(500);
                  Serial.print(".");
                }
               
                // Configure WiFiClientSecure with AWS IoT device credentials
                net.setCACert(AWS_CERT_CA);
                net.setCertificate(AWS_CERT_CRT);
                net.setPrivateKey(AWS_CERT_PRIVATE);
               
                // Connect to MQTT broker on AWS endpoint
                client.setServer(AWS_IOT_ENDPOINT, 8883);
               
                // Set up message handler
                client.setCallback(messageHandler);
               
                // Print connection status message
                Serial.println("Connecting to AWS IoT");
               
                // Attempt to connect to AWS IoT Core
                while (!client.connect(THINGNAME)) {
                  Serial.print(".");
                  delay(100);
                }
               
                // Check if connection is successful
                if (!client.connected()) {
                  Serial.println("AWS IoT Timeout!");
                  return;
                }
               
                // Subscribe to a topic on AWS IoT Core
                client.subscribe(AWS_IOT_SUBSCRIBE_TOPIC);
               
                // Print success message
                Serial.println("AWS IoT Connected!");
              }
              
              // Function to publish a message to AWS IoT Core
              void publishMessageA(String id, char __typename[], char createdAt[], char username[], char description[], float humidity,float light,float moisture,float smell,float temperature, char name[]) {
                // Create a JSON document
                StaticJsonDocument<600> doc;
                doc["id"] = id;
                doc["__typename"] = __typename;
                doc["createdAt"] = createdAt;
                doc["username"] = username;
                doc["description"] = description;
                doc["humidity"] = humidity;
                doc["light"] = light;
                doc["moisture"] = moisture;
                doc["smell"] = smell;
                doc["temperature"] = temperature;
                doc["name"] = name;
              
                // Serialize JSON document to a buffer
                char jsonBuffer[2048];
                serializeJson(doc, jsonBuffer);
               
                // Publish message to AWS IoT Core
                client.publish(AWS_IOT_PUBLISH_TOPIC, jsonBuffer);
              }
