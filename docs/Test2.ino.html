              const int SOIL_MOISTURE_PIN = 34;

              // Device Information
              char __typename[] = "Note";
              char username[] = "cameron";
              char description[] = "personal plant";
              char name[] = "orchid";
              char updatedAt[] = "2024-04-28T18:43:32.293Z";
              float smell = 95;
              
              // Web Server
              AsyncWebServer server(80);
              
              // Sensors
              BH1750 lightMeter;
              Adafruit_SGP40 sgp;
              Adafruit_SHT31 sht31 = Adafruit_SHT31();
              
              // NTP Client
              WiFiUDP ntpUDP;
              NTPClient timeClient(ntpUDP, "pool.ntp.org", 0 * 3600);
              
              // Handle root endpoint
              void handleRoot(AsyncWebServerRequest *request) {
                request->redirect("/captive-portal");
              }
              
              // Handle captive portal page
              void handleCaptivePortal(AsyncWebServerRequest *request) {
                request->send(200, "text/plain", "This is the captive portal page");
              }
              
              // Generate random ID
              String generateRandomID(int length) {
                const char charset[] = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
                String randomID = "";
                for (int i = 0; i < length; ++i) {
                  randomID += charset[random(0, sizeof(charset) - 1)]; // Adjusted range for random function
                }
                return randomID;
              }
              
              // Setup function
              void setup() {
                // Start serial communication
                Serial.begin(115200);
              
                // Initialize random seed
                randomSeed(analogRead(0));
              
                // Initialize I2C communication
                Wire.begin();
              
                // Initialize light sensor
                lightMeter.begin();
              
                // Initialize temperature and humidity sensor
                sht31.begin(0x44);
                
                // Configure WiFi
                WiFiManager wifiManager;
                if (!wifiManager.autoConnect("ESP32AP")) {
                  Serial.println("Failed to connect to saved WiFi network");
                  if (!wifiManager.startConfigPortal("ESP32AP")) {
                    Serial.println("Failed to connect and hit timeout");
                    delay(3000);
                    ESP.restart();
                    delay(5000);
                  }
                }
              
                // Initialize NTP Client
                timeClient.begin();
                timeClient.update();
              
                // Connect to AWS
                connectAWS();
              
                // Setup web server endpoints
                server.on("/", HTTP_GET, handleRoot);
                server.on("/captive-portal", HTTP_GET, handleCaptivePortal);
                server.begin();
              
                // Initialize SGP40 sensor
                if (!sgp.begin()) {
                  Serial.println("SGP40 sensor not found");
                }
              
                // Indicate successful WiFi connection
                Serial.println("Connected to WiFi!");
              }
              
              // Loop function
              void loop() {
                // Generate random ID
                String id = generateRandomID(20);
                Serial.print("id: ");
                Serial.println(id);
                char dateTime[26];
                timeClient.update();
              
                // Get current date and time
                char currentDate[11];
                time_t now = timeClient.getEpochTime();
                struct tm *timeinfo;
                timeinfo = localtime(&now);
                sprintf(currentDate, "%04d-%02d-%02d", timeinfo->tm_year + 1900, timeinfo->tm_mon + 1, timeinfo->tm_mday);
              
                // Get current time
                char currentTime[13];
                sprintf(currentTime, "%02d:%02d:%02d.%03d", timeinfo->tm_hour, timeinfo->tm_min, timeinfo->tm_sec, millis() % 1000);
                sprintf(dateTime, "%sT%sZ", currentDate, currentTime);
              
                // Print metadata
                Serial.print("__typename: ");
                Serial.println(__typename);
                Serial.print("createdAt: ");
                Serial.println(dateTime);
                
                // Read sensor data
                float light = lightMeter.readLightLevel();
                float temperature = (sht31.readTemperature() * 9/5) + 32;
                float humidity = sht31.readHumidity();
                float temp = sgp.measureVocIndex(temperature,humidity);
                if(temp > 95) {
                  smell = temp;
                }
                float moisture = map(analogRead(SOIL_MOISTURE_PIN), 3500, 1400, 0, 100);
                
                // Print sensor data
                Serial.print("Soil Moisture: "); Serial.println(moisture);
                Serial.print("Light: "); Serial.print(light); Serial.println(" lx");
                Serial.print("Temp *F = "); Serial.print(temperature); Serial.print("\t\t");
                Serial.print("Hum. % = "); Serial.println(humidity);
                Serial.println("VOC = "); Serial.println(smell);
              
                // Print metadata
                Serial.print("updatedAt: ");
                Serial.println(updatedAt);
                Serial.print("name: ");
                Serial.println(name);
              
                // Publish data to AWS
                publishMessageA(id, __typename, dateTime, username, description, humidity, light, moisture, smell, temperature, name);
                client.loop();
                
                // Delay for stability
                delay(1000); // Adjust delay as needed
              }
